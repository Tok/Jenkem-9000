36 class IrcConnector(val eventRouter: EventRouter) extends GridLayout {
37   val networkCombo = createCombo(IrcSettings.getDefaultNetwork.net)
38   val portCombo = createCombo(IrcSettings.getDefaultNetwork.port)
39   val delayBox = new TextField
40   delayBox.setValue(IrcSettings.defaultDelayMs.toString)
41   delayBox.setWidth("50px")
42
43   if (OpenShiftUtil.isOnOpenshift) {
44     portCombo.setReadOnly(true)
45     delayBox.setReadOnly(true)
46   } else {
47     //more options if application is not running on OpenShift
48     IrcSettings.networks.foreach(n => if (n.location.equals(IrcSettings.Loc.ELSEWHERE)) {
49         networkCombo.addItem(n.net)
50         portCombo.addItem(n.port)
51     })
52   }
53
54   networkCombo.addValueChangeListener(new Property.ValueChangeListener {
55     override def valueChange(event: ValueChangeEvent): Unit = {
56       IrcSettings.networks.find(_.net.equalsIgnoreCase(networkCombo.getValue.toString)) match {
57         case Some(n) => portCombo.select(n.port)
58         case _ => Unit
59       }
60     }
61   })
62
63   delayBox.addValueChangeListener(new Property.ValueChangeListener {
64     override def valueChange(event: ValueChangeEvent): Unit = {
65       try {
66         val delay = delayBox.getValue.toInt
67         if (delay < IrcSettings.warningDelayMs) {
68           Notification.show("Warning: Delay is low. Please make sure you are not flooding, or the bot might get kicked.")
69         } else if (delay > IrcSettings.maxDelayMs) {
70           Notification.show("Warning: Delay is very high. Please make sure you know what you are doing.")
71         }
72         IrcService.setDelay(delay)
73       } catch {
74         case e: NumberFormatException =>
75           Notification.show("Delay must be a number. Value has been reset to default.")
76           delayBox.setValue(IrcSettings.defaultDelayMs.toString)
77       }
78     }
79   })
80
81   val width = "400px"
82   val compWidth = "250px"
83
84   setWidth(width)
85
86   setRows(6)
87   setColumns(3)
88
89   //   0       1         2
90   //0 [Label ][ComboBox][ComboBox]
91   //1 [Label ][Box               ]
92   //2 [Label ][Box               ]
93   //3 [Label ][ComboBox          ]
94   //4 [Label ][Buttons           ]
95   //5 [Label ][Status  ][Button  ]
96
97   val networkCaption = new Label("IRC Network: ")
98   networkCaption.setWidth("150px")
99   networkCombo.setWidth("180px")
100   portCombo.setWidth("70px")
101   delayBox.setWidth("100px")
102   val channelCaption = new Label("Channel: ")
103   val channelBox = new TextField
104   channelBox.setWidth(compWidth)
105   val nickCaption = new Label("Nick: ")
106   val nickBox = new TextField
107   nickBox.setWidth(compWidth)
108   val actionCaption = new Label("Actions: ")
109   val buttonLayout = new GridLayout
110   val connectButton = new Button("Connect")
111   val disconnectButton = new Button("Disconnect")
112   val sendButton = new Button("Send Conversion")
113   val delayCaption = new Label("Delay (ms): ")
114   val statusCaption = new Label("Status: ")
115   val statusLayout = new VerticalLayout
116   val statusLabel = new Label("")
117   statusLabel.setWidth(compWidth)
118   val refreshButton = new Button("Refresh Bot Status")
119
120   //addComponent(component, column1, row1, column2, row2)
121   addComponent(networkCaption, 0, 0)
122   addComponent(networkCombo, 1, 0)
123   addComponent(portCombo, 2, 0)
124   addComponent(channelCaption, 0, 1)
125   addComponent(channelBox, 1, 1, 2, 1)
126   addComponent(nickCaption, 0, 2)
127   addComponent(nickBox, 1, 2, 2, 2)
128   addComponent(delayCaption, 0, 3)
129   addComponent(delayBox, 1, 3, 2, 3)
130   addComponent(actionCaption, 0, 4)
131   addComponent(buttonLayout, 1, 4, 2, 4)
132   addComponent(statusCaption, 0, 5)
133   addComponent(statusLayout, 1, 5, 2, 5)
134   buttonLayout.setRows(2)
135   buttonLayout.setColumns(2)
136   buttonLayout.addComponent(connectButton, 0, 0)
137   buttonLayout.addComponent(disconnectButton, 1, 0)
138   buttonLayout.addComponent(sendButton, 0, 1, 1, 1)
139   statusLayout.addComponent(statusLabel)
140   statusLayout.addComponent(refreshButton)
141
142   channelBox.setValue(IrcSettings.defaultChannel)
143   nickBox.setValue(IrcSettings.defaultNick)
144   refresh
145
146   private def createCombo(item: Any): ComboBox = {
147     val combo = new ComboBox
148     combo.addItem(item)
149     combo.select(item)
150     combo.setTextInputAllowed(true)
151     combo.setNewItemsAllowed(true)
152     combo.setNullSelectionAllowed(false)
153     combo.setImmediate(true)
154     combo
155   }
156
157   connectButton.addClickListener(new Button.ClickListener {
158     override def buttonClick(event: ClickEvent): Unit = {
159       Notifications.showConnectToIrc(Page.getCurrent)
160       connectButton.setEnabled(false)
161       val network = networkCombo.getValue.toString
162       val port = Integer.parseInt(portCombo.getValue.toString)
163       val channel = channelBox.getValue
164       val nick = nickBox.getValue
165       IrcService.setDelay(delayBox.getValue.toInt)
166       val message = IrcService.connect(network, port, channel, nick)
167       statusLabel.setValue(message)
168     }
169   })
170
171   disconnectButton.addClickListener(new Button.ClickListener {
172     override def buttonClick(event: ClickEvent): Unit = {
173       disconnectButton.setEnabled(false)
174       statusLabel.setValue(IrcService.disconnect)
175     }
176   })
177
178   sendButton.addClickListener(new Button.ClickListener {
179     override def buttonClick(event: ClickEvent): Unit = {
180       sendButton.setEnabled(false)
181       statusLabel.setValue("Sending...")
182       eventRouter.fireEvent(new SendToIrcEvent)
183     }
184   })
185
186   refreshButton.addClickListener(new Button.ClickListener {
187     override def buttonClick(event: ClickEvent): Unit = refresh
188   })
189
190   def sendToIrc(message: java.util.List[String]): Unit = {
191     IrcService.sendMessage(message)
192     Notifications.showPlayToIrc(Page.getCurrent)
193     refresh
194   }
195
196   def refresh: Unit = showBotStatus(IrcService.getBotStatus)
197
198   private def showBotStatus(stat: BotStatus.Value): Unit = {
199     if (stat.isConnected || stat.isSending) {
200       if (stat.isSending) { statusLabel.setValue("Bot is busy...") }
201       else { statusLabel.setValue("Bot is connected.") }
202       networkCombo.setValue(stat.network)
203       networkCombo.setEnabled(false)
204       portCombo.setEnabled(false)
205       channelBox.setValue(stat.channel)
206       channelBox.setEnabled(false)
207       nickBox.setValue(stat.name)
208       nickBox.setEnabled(false)
209       delayBox.setValue(stat.delay.toString)
210       delayBox.setEnabled(false)
211       connectButton.setEnabled(false)
212       disconnectButton.setEnabled(true)
213       sendButton.setEnabled(!stat.isSending)
214     } else {
215       statusLabel.setValue("Bot is not connected.")
216       networkCombo.setEnabled(true)
217       portCombo.setEnabled(true)
218       channelBox.setEnabled(true)
219       nickBox.setEnabled(true)
220       delayBox.setEnabled(true)
221       connectButton.setEnabled(true)
222       disconnectButton.setEnabled(false)
223       sendButton.setEnabled(false)
224     }
225   }
