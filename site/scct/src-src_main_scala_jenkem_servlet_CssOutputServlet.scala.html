14 package jenkem.servlet
15
16 import javax.servlet.http.HttpServletRequest
17 import javax.servlet.http.HttpServletResponse
18
19 /**
20 * Servlet to retrieve and return stored CSS.
21 */
22 class CssOutputServlet extends AbstractOutputServlet {
23   override def doGet(request: HttpServletRequest, response: HttpServletResponse): Unit = {
24     response.setCharacterEncoding(encoding)
25     response.setContentType("text/css")
26     jenkemService.getImageCssByName(obtainName(request)) match {
27       case Some(imageCss) => response.getWriter.write(imageCss.css)
28       case None => response.getWriter.write("Fail: CSS couldn't be obtained.")
29     }
30   }
31 }
