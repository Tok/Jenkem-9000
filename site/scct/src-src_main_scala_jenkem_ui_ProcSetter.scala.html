16 import scala.collection.immutable.HashMap
17
18 import com.vaadin.data.Property
19 import com.vaadin.data.Property.ValueChangeEvent

29 import jenkem.engine.Setting

32 class ProcSetter(val eventRouter: EventRouter) extends GridLayout {
33   var triggeringDisabled = true
34   setWidth("250px")
35
36   val vcl = new Property.ValueChangeListener {
37     override def valueChange(event: ValueChangeEvent): Unit = {
38       if (!triggeringDisabled) {
39         eventRouter.fireEvent(new DoConversionEvent(false, false))
40       }
41     }
42   }
43
44   type PS = Setting
45   val settings = Setting.values
46   setRows(3)
47   setColumns(3)
48
49   val boxes = makeComponentMap[CheckBox](settings, makeBox)
50   val labels = makeComponentMap[Label](settings.take(2), makeLabel)
51   val sliders = makeComponentMap[Slider](settings.take(2), makeSlider)
52   (0 to 1).foreach(makeSliderRow(_))
53
54   val layout = new HorizontalLayout
55   (2 to 5).foreach(i => layout.addComponent(boxes.get(settings(i)).get))
56   addComponent(layout, 0, 2, 2, 2)
57
58   triggeringDisabled = false
59
60   private def makeComponentMap[T](sets: List[PS], f: PS => Component): HashMap[PS, T] = {
61     def makeComponentMap0(hm: HashMap[PS, T], s: List[PS]): HashMap[PS, T] = {
62       if (s.isEmpty) { hm } else {
63         val comp = f(s.head).asInstanceOf[T]
64         makeComponentMap0(hm + ((s.head, comp)), s.tail)
65       }
66     }
67     makeComponentMap0(new HashMap[PS, T], sets)
68   }
69
70   private def makeBox(setting: PS): CheckBox = {
71     val box = new CheckBox(setting.caption)
72     box.setWidth("50px")
73     box.setImmediate(true)
74     box.addValueChangeListener(vcl)
75     box
76   }
77
78   private def makeLabel(setting: PS): Label = {
79     val label = new Label(Setting.default.toString)
80     label.setWidth("30px")
81     label.setStyleName("sliderLabel")
82     label
83   }
84
85   private def makeSlider(setting: PS): Slider = {
86     val slider = new Slider(None.orNull)
87     slider.setWidth("170px")
88     slider.setMin(Setting.min)
89     slider.setMax(Setting.max)
90     slider.setValue(Setting.default)
91     slider.setImmediate(true)
92     slider.addValueChangeListener(new Property.ValueChangeListener {
93       override def valueChange(event: ValueChangeEvent): Unit = {
94         labels.get(setting).get.setValue("%1.0f".format(event.getProperty.getValue))
95         if (!triggeringDisabled) {
96           eventRouter.fireEvent(new DoConversionEvent(false, false))
97         }
98       }
99     })
100     slider
101   }
102
103   private def makeSliderRow(i: Int): Unit = {
104     val setting = settings(i)
105     addAligned(boxes.get(setting).get, 0, i)
106     addAligned(sliders.get(setting).get, 1, i)
107     addAligned(labels.get(setting).get, 2, i)
108   }
109
110   private def addAligned(comp: Component, row: Int, i: Int): Unit = {
111     addComponent(comp, row, i)
112     setComponentAlignment(comp, Alignment.MIDDLE_LEFT)
113   }
114
115   def reset(hasAnsi: Boolean): Unit = {
116     triggeringDisabled = true
117     val vals = Setting.getInitial(hasAnsi)
118     settings.foreach(set(_))
119     def set(setting: PS): Unit = {
120       boxes.get(setting).get.setValue(vals.has(setting))
121       sliders.get(setting) match {
122         case Some(s) => s.setValue(vals.get(setting))
123         case None => { }
124       }
125     }
126     triggeringDisabled = false
127   }
128
129   def getSettings: Setting.Instance = {
130     def makePair(setting: PS): Setting.Pair = {
131       (boxes.get(setting).get.getValue,
132       sliders.get(setting).get.getValue.intValue)
133     }
134     def makeBool(setting: PS): Boolean = {
135       boxes.get(setting).get.getValue
136     }
137     new Setting.Instance(
138       makePair(Setting.UPDOWN), makePair(Setting.LEFTRIGHT),
139       makeBool(Setting.DBQP), makeBool(Setting.DIAGONAL),
140       makeBool(Setting.VERTICAL), makeBool(Setting.HORIZONTAL))
141   }
142 }
