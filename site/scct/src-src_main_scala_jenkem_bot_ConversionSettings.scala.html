16 import java.util.HashMap

29 class ConversionSettings {
30   @BeanProperty var width: Int = _
31   @BeanProperty var method: Method = _
32   @BeanProperty var colorMap: Color.IrcMap = _
33   @BeanProperty var schemeName: String = _
34   @BeanProperty var chars: String = _
35   @BeanProperty var kick: Kick = _
36   @BeanProperty var power: Power = _
37   @BeanProperty var proportion: Proportion = _
38
39   reset
40
41   def reset: Unit = {
42     width = InitUtil.DEFAULT_WIDTH
43     method = Method.default
44     colorMap = Scheme.createColorMap(Scheme.default)
45     schemeName = Scheme.default.name
46     chars = Pal.default.chars
47     kick = Kick.OFF
48     power = Power.default
49     proportion = Proportion.default
50   }
51
52   def getParams(imageData: Color.RgbMap): Engine.Params = {
53     val (methodOpt, scheme, charsetOpt) = InitUtil.getDefaults(imageData)
54     methodOpt match {
55       case Some(meth) =>
56         if (method.hasColor != meth.hasColor && !method.equals(Method.Pwntari)) {
57           //if method is Pwntari it should not be changed here
58           //because it uses different scaling.
59           method = meth
60         }
61       case None => { }
62     }
63     if (scheme.equals(Scheme.Bwg)) {
64       colorMap = Scheme.createColorMap(scheme)
65       schemeName = scheme.name
66     }
67     charsetOpt match {
68       case Some(chrset) =>
69         if (!method.hasColor) {
70           if (Pal.hasAnsi(chars)) { chars = chrset.chars }
71         } else if (method.equals(Method.Pwntari)) {
72           chars = "▄"
73         }
74       case None => { }
75     }
76     new Engine.Params(
77         method,
78         imageData,
79         colorMap,
80         chars.replaceAll("[,0-9]", ""),
81         Setting.getInitial(Pal.hasAnsi(chars)),
82         0,
83         0,
84         power
85     )
86   }
87 }
