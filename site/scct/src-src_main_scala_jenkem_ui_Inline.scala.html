14 package jenkem.ui
15
16 import com.vaadin.server.Resource
17 import com.vaadin.shared.ui.label.ContentMode
18 import com.vaadin.ui.Alignment
19 import com.vaadin.ui.Button
20 import com.vaadin.ui.Button.ClickEvent
21 import com.vaadin.ui.GridLayout
22 import com.vaadin.ui.Image
23 import com.vaadin.ui.Label
24 import com.vaadin.ui.VerticalLayout
25 import java.awt.image.BufferedImage
26 import jenkem.util.AwtImageUtil
27 import jenkem.engine.Method
28
29 class Inline extends VerticalLayout {
30   val spacer = new Label
31   spacer.setWidth(500 + "px")
32   spacer.setHeight(0 + "px")
33
34   val previewLayout = new GridLayout
35   previewLayout.setRows(2)
36   previewLayout.setColumns(1)
37   previewLayout.addComponent(spacer, 0, 0)
38   previewLayout.setComponentAlignment(spacer, Alignment.TOP_LEFT)
39   addComponent(previewLayout)
40   setComponentAlignment(previewLayout, Alignment.TOP_LEFT)
41
42   val preview = new Label
43   preview.setContentMode(ContentMode.HTML)
44
45   val image = new Image
46
47   val overlayButton = new Button
48   overlayButton.setImmediate(true)
49   overlayButton.addClickListener(new Button.ClickListener {
50     override def buttonClick(event: ClickEvent): Unit = { switchContent }
51   })
52   addComponent(overlayButton)
53   setComponentAlignment(overlayButton, Alignment.TOP_LEFT)
54
55   var isShowHtml = false //gets switched on load
56   private def switchContent: Unit = {
57     previewLayout.removeComponent(0, 1)
58     previewLayout.addComponent(if (isShowHtml) { image } else { preview }, 0, 1)
59     overlayButton.setCaption(if (isShowHtml) { "Show HTML" } else { "Show Image" })
60     isShowHtml = !isShowHtml
61   }
62
63   switchContent
64
65   def setIntermediate(img: BufferedImage, method: Method): Unit = {
66     val widthFactor = if (!method.equals(Method.Pwntari)) { 3 } else { 6 }
67     val intermediate = AwtImageUtil.getScaled(img, img.getWidth * widthFactor, img.getHeight * 6)
68     val resource = AwtImageUtil.makeVaadinResource(intermediate, "Intermediate")
69     image.setSource(resource)
70   }
71   def setValue(value: String): Unit = preview.setValue(value)
72   def reset: Unit = if (!isShowHtml) { switchContent }
73 }
