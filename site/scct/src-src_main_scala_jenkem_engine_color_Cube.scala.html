14 package jenkem.engine.color

16 import scala.Array.canBuildFrom
17 import jenkem.engine.Pal

19 object Cube {
20   def getNearest(col: Color.Rgb, colorMap: Color.IrcMap): Short = {
21     val list: List[WeightedColor] = makeWeightedList(col, colorMap, Scheme.ircColors, Nil)
22     val shuffled: List[WeightedColor] = util.Random.shuffle(list)
23     val ordered: List[WeightedColor] = shuffled.sortBy(_.weight)
24     ordered.head.ircColor.irc
25   }
26
27   private def makeWeightedList(col: Color.Rgb, colorMap: Color.IrcMap,
28       ic: List[Scheme.IrcColor], wl: List[WeightedColor]): List[WeightedColor] = {
29     if (ic.isEmpty) { wl }
30     else { makeWeightedList(col, colorMap, ic.tail, createWc(colorMap, col, ic.head) :: wl) }
31   }
32
33   def getTwoNearestColors(col: Color.Rgb, colorMap: Color.IrcMap, power: Power): Color = {
34     def calcPoweredStrength(wc: WeightedColor, p: Float): Float = {
35       Math.pow(calcStrength(col, wc.ircColor.rgb, colorMap.get(wc.ircColor).get.floatValue), p).toFloat
36     }
37
38     val list: List[WeightedColor] = makeWeightedList(col, colorMap, Scheme.ircColors, Nil)
39
40     // shuffle to balance colors with same strength
41     val shuffled: List[WeightedColor] = util.Random.shuffle(list)
42     val ordered: List[WeightedColor] = shuffled.sortBy(_.weight)
43     val strongest: WeightedColor = ordered.head
44     val second: WeightedColor = ordered.tail.head
45
46     // strength is used to calculate which character to return.
47     // TODO XXX FIXME tune and explain this, #math suggested Voronoi diagrams.
48     // ..using a variable for power and the formula applied in calcStrength
49     // is just an approximative workaround. instead there should be a
50     // mathematically provable correct way on how to weigh two colors
51     // against each other in regard to their distance to the
52     // center of the cube 127,127,127
53
54     val p: Float = power.exponent
55     val strongestStrength = calcPoweredStrength(strongest, p)
56     val secondStrength = calcPoweredStrength(second, p)
57     val strength: Float = strongestStrength / secondStrength
58
59     new Color(
60       second.ircColor.irc.toString, second.ircColor.rgb,
61       strongest.ircColor.irc.toString, strongest.ircColor.rgb,
62       strength)
63   }
64
65   private def calcStrength(col: Color.Rgb, comp: Color.Rgb, factor: Float): Float = {
66     def calcDistance(from: Color.Rgb, to: Color.Rgb): Float = {
67       Math.sqrt((Math.pow(to._1 - from._1, 2D)
68         + Math.pow(to._2 - from._2, 2D)
69         + Math.pow(to._3 - from._3, 2D))).toFloat
70     }
71     calcDistance(col, comp) / factor
72   }
73
74   private def createWc(colorMap: Color.IrcMap,
75     col: (Short, Short, Short), ic: Scheme.IrcColor): WeightedColor = {
76     val weight = calcStrength(col, (ic.rgb._1, ic.rgb._2, ic.rgb._3), colorMap.get(ic).get.floatValue)
77     new WeightedColor(ic, weight)
78   }
79
80   def getColorChar(colorMap: Color.IrcMap,
81     charset: String, p: Power, rgb: (Short, Short, Short)): String = {
82     val c: Color = getTwoNearestColors(rgb, colorMap, p)
83     c.fg + "," + c.bg + Pal.getChar(charset, c.strength)
84   }
85 }
