21 object InitUtil {
22   val DEFAULT_WIDTH = 68
23   val MIN_WIDTH = 16
24   val MAX_WIDTH = 74
25
26   val COLOR_TOLERANCE: Short = 15 //arbitrary value (absolute RGB)
27   val BLACK_LIMIT: Short = 30
28   val WHITE_LIMIT: Short = 225
29
30   def calculateNewSize(method: Method, lineWidth: Int, originalWidth: Int, originalHeight: Int): (Int, Int) = {
31     val width = if (!method.equals(Method.Pwntari)) { lineWidth * 2 } else { lineWidth }
32     val height = lineWidth * originalHeight / originalWidth
33     (Math.max(1, width), Math.max(1, height))
34   }
35
36   def calculateProportionalSize(method: Method, lineWidth: Int, oWidth: Int, oHeight: Int): (Int, Int) = {
37     def depend(p: (Int, Int)): (Int, Int) = {
38       if (!method.equals(Method.Pwntari)) { (Math.max(1, p._1 * 2), Math.max(1, p._2)) }
39       else { (Math.max(1, p._1), Math.max(1, p._2)) }
40     }
41     if (oWidth <= lineWidth) {
42       val ratio: Int = lineWidth / oWidth
43       depend((oWidth * ratio, oHeight * ratio))
44     } else {
45       val range = ((lineWidth / 2) to lineWidth)
46       val wMods = range.filter(oWidth % _ == 0)
47       val hMods = range.filter(oHeight % _ == 0)
48       val both = wMods.filter(hMods.contains(_))
49       if (!both.isEmpty) {
50         val divisor = oWidth / both.last
51         depend((oWidth / divisor, oHeight / divisor))
52       } else if (!wMods.isEmpty) {
53         val divisor = oWidth / wMods.last
54         depend((oWidth / divisor, oHeight / divisor))
55       } else {
56         calculateNewSize(method, lineWidth, oWidth, oHeight)
57       }
58     }
59   }
60
61   def getDefaults(imageRgb: Color.RgbMap): (Option[Method], Scheme, Option[Pal.Charset]) = {
62     val grey = imageRgb.map(pixel => isPixelGray(pixel._2)).toList
63     val bw = imageRgb.map(pixel => isPixelBlackOrWhite(pixel._2)).toList
64     val gRatio: Double = grey.filter(_ == true).length.doubleValue / grey.length.doubleValue
65     val bwRatio: Double = bw.filter(_ == true).length.doubleValue / bw.length.doubleValue
66     val scheme = if (gRatio > 0.9D) { Scheme.Bwg } else { Scheme.default }
67     if (bwRatio > 0.9D) {
68       (Some(Method.Stencil), scheme, Some(Pal.HCrude))
69     } else {
70       (None, scheme, None)
71     }
72   }
73
74   private def isPixelGray(rgb: Color.Rgb): Boolean = {
75     ((Math.abs(rgb._1 - rgb._2) <= COLOR_TOLERANCE)
76       && (Math.abs(rgb._1 - rgb._3) <= COLOR_TOLERANCE)
77       && (Math.abs(rgb._2 - rgb._3) <= COLOR_TOLERANCE))
78   }
79
80   private def isPixelBlackOrWhite(rgb: Color.Rgb): Boolean = {
81     val mean = (rgb._1 + rgb._2 + rgb._3).doubleValue / 3
82     mean < BLACK_LIMIT || mean > WHITE_LIMIT
83   }
