34 class GalleryTab(val eventRouter: EventRouter) extends VerticalLayout {
35   setCaption("Gallery")

40   sealed abstract class Column(val name: String, val c: Class[_], val align: Align)
41   case object ICON extends Column("Icon", classOf[Link], Table.Align.CENTER)
42   case object NAME extends Column("Name", classOf[Link], Table.Align.LEFT)
43   case object METHOD extends Column("Method", classOf[String], Table.Align.LEFT)
44   case object CHARS extends Column("Characters", classOf[String], Table.Align.LEFT)
45   case object CONT extends Column("Contrast", classOf[Int], Table.Align.RIGHT)
46   case object BRIGHT extends Column("Brightness", classOf[Int], Table.Align.RIGHT)
47   case object LINES extends Column("Lines", classOf[Int], Table.Align.RIGHT)
48   case object WIDTH extends Column("Width", classOf[Int], Table.Align.RIGHT)
49   case object CREATION extends Column("Creation", classOf[String], Table.Align.LEFT)
50   val cols = Array(ICON, NAME, METHOD, CHARS, CONT, BRIGHT, LINES, WIDTH, CREATION)
51   val names: Array[Object] = cols.map(_.name.toLowerCase)
52
53   val table = new Table
54   table.setWidth("100%")
55   table.setHeight("1000px")
56   table.setSelectable(false)
57   table.setMultiSelect(false)
58   table.setImmediate(true)
59   table.setColumnReorderingAllowed(true)
60   table.setColumnCollapsingAllowed(true)
61
62   class ImageInfoBean(info: ImageInfo) {
63     def getIcon: Link = {
64       val link = new Link
65       link.setTargetName("_blank")
66       val ex = new ExternalResource(HtmlUtil.getHtmlUrl(info.name))
67       link.setResource(ex)
68       val buffered = AwtImageUtil.decodeFromBase64(info.icon)
69       val resource = AwtImageUtil.makeVaadinResource(buffered, info.name)
70       link.setIcon(resource)
71       link
72     }
73     def getName: Link = {
74       val ex = new ExternalResource(HtmlUtil.getHtmlUrl(info.name))
75       val link = new Link(info.name, ex)
76       link.setTargetName("_blank")
77       link
78     }
79     def getMethod: String = info.method
80     def getCharacters: String = info.characters
81     def getContrast: Integer = info.contrast
82     def getBrightness: Integer = info.brightness
83     def getLines: Integer = info.lines
84     def getWidth: Integer = info.lineWidth
85     def getCreation: String = info.creation
86   }
87
88   cols.foreach(col => table.addContainerProperty(col.name.toLowerCase, col.c, classOf[String]))
89   val bic = new BeanItemContainer[ImageInfoBean](classOf[ImageInfoBean])
90   val namesColl: java.util.Collection[Object] = names.toSeq
91   table.setContainerDataSource(bic, namesColl)
92   cols.foreach(col => table.setColumnAlignment(col.name.toLowerCase, col.align))
93   cols.foreach(col => table.setColumnHeader(col.name.toLowerCase, col.name))
94
95   addComponent(table)
96
97   update
98
99   def update: Unit = {
100     bic.removeAllItems
101     PersistenceService.getAllImageInfo match {
102       case Some(info) => info.foreach(info => bic.addItem(new ImageInfoBean(info)))
103       case None => { Notifications.showDbNotConnected(Page.getCurrent) }
104     }
105   }
