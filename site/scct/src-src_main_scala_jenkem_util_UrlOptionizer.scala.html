14 package jenkem.util
15
16 import java.net.MalformedURLException
17
18 object UrlOptionizer {
19   object UrlExtractor {
20     def unapply(u: java.net.URL): Some[(String, String, Int, String)] =
21       Some((u.getProtocol, u.getHost, u.getPort, u.getPath))
22   }
23   def extract(link: String): Option[java.net.URL] = {
24     try {
25       val u = new java.net.URL(link)
26       u match {
27         case UrlExtractor(protocol, host, port, path) => Some(u)
28       }
29     } catch {
30       case murle: MalformedURLException => None
31     }
32   }
33 }
