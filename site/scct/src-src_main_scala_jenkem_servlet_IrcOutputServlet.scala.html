14 package jenkem.servlet
15
16 import jenkem.persistence.data.ImageIrc
17 import javax.jdo.annotations.PersistenceCapable
18 import javax.servlet.http.HttpServletRequest
19 import javax.servlet.http.HttpServletResponse
20
21 /**
22 * Servlet to retrieve and return stored IRC output.
23 */
24 class IrcOutputServlet extends AbstractOutputServlet {
25   override def doGet(request: HttpServletRequest, response: HttpServletResponse): Unit = {
26     response.setCharacterEncoding(encoding)
27     response.setContentType("text/plain")
28     jenkemService.getImageIrcByName(obtainName(request)) match {
29       case Some(imageIrc) =>
30         Option(imageIrc.irc) match {
31           case Some(irc) => response.getWriter.write(irc)
32           case None => response.getWriter.write("Fail: IRC text is empty.")
33         }
34       case None => response.getWriter.write("Fail: IRC text couldn't be obtained.")
35     }
36   }
37 }
