14 package jenkem.persistence
15
16 import java.util.Properties
17
18 import javax.jdo.JDOHelper
19 import javax.jdo.PersistenceManagerFactory
20
21 /**
22 * Holds a single instance of the PersistenceManagerFactory.
23 */
24 object PMF {
25   final val dbHost = System.getenv("OPENSHIFT_MONGODB_DB_HOST")
26   final val dbPort = System.getenv("OPENSHIFT_MONGODB_DB_PORT")
27   final val dbName = System.getenv("OPENSHIFT_APP_NAME")
28   final val dbUser = System.getenv("OPENSHIFT_MONGODB_DB_USERNAME")
29   final val dbPass = System.getenv("OPENSHIFT_MONGODB_DB_PASSWORD")
30
31   final val properties: Properties = new Properties
32   //properties.setProperty("datanucleus.cache.level2.type", "xmemcached")
33   properties.setProperty("javax.jdo.PersistenceManagerFactoryClass", "org.datanucleus.api.jdo.JDOPersistenceManagerFactory")
34   properties.setProperty("javax.jdo.option.Mapping", "mongodb")
35
36   Option(dbHost) match {
37     case Some(v) if (!v.isEmpty) =>
38       properties.setProperty("javax.jdo.option.ConnectionURL", "mongodb:" + v + ":" + dbPort + "/" + dbName)
39       properties.setProperty("javax.jdo.option.ConnectionUserName", dbUser)
40       properties.setProperty("javax.jdo.option.ConnectionPassword", dbPass)
41     case None =>
42       properties.setProperty("javax.jdo.option.ConnectionURL", "mongodb:/jenkem")
43   }
44
45   final val pmf: PersistenceManagerFactory = JDOHelper.getPersistenceManagerFactory(properties)
46   def get: PersistenceManagerFactory = pmf
47 }
