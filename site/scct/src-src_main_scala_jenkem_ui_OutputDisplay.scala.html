14 package jenkem.ui
15
16 import com.vaadin.event.FieldEvents.FocusEvent
17 import com.vaadin.event.FieldEvents.FocusListener
18 import com.vaadin.ui.Label
19 import com.vaadin.ui.TextArea
20 import com.vaadin.ui.VerticalLayout
21
22 class OutputDisplay extends VerticalLayout {
23   setSpacing(true)
24   setWidth("400px")
25
26   val outputCaptionLabel = new Label("Binary Output For IRC: ")
27   val ircText = new TextArea
28   ircText.setWidth("400px")
29   ircText.setWordwrap(false)
30   //ircText.setReadOnly(true)
31   ircText.addFocusListener(new FocusListener {
32     override def focus(event: FocusEvent): Unit = ircText.selectAll
33   })
34   addComponent(outputCaptionLabel)
35   addComponent(ircText)
36
37   def addIrcOutput(ircOutput: List[String]): Unit = {
38     val lines = new StringBuilder(ircOutput.size)
39     ircOutput.foreach(line => lines.append(line))
40     ircText.setValue(lines.toString)
41   }
42 }
