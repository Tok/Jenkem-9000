32 /**
33 * Implementation of service to handle the persistence of reports.
34 */
35 object PersistenceService {
36   val QUERY_RANGE = 200L
37
38   def saveJenkemImage(jenkemImage: JenkemImage): Boolean = {
39     synchronized {
40       val pm = PMF.get.getPersistenceManager
41       val tx: Transaction = pm.currentTransaction
42       try {
43         val name = jenkemImage.info.name
44         getByName[ImageInfo](name, classOf[ImageInfo]) match {
45           case Some(t) =>
46             tx.begin
47             jenkemImage.values.foreach(part => pm.deletePersistent(pm.getObjectById(part.c, name)))
48             tx.commit
49           case None => { }
50         }
51         tx.begin
52         pm.makePersistent(jenkemImage.info)
53         pm.makePersistent(jenkemImage.html)
54         pm.makePersistent(jenkemImage.css)
55         pm.makePersistent(jenkemImage.irc)
56         tx.commit
57         true
58       } catch {
59         case me: MongoException => false
60         case jdodse: JDODataStoreException => false
61       } finally {
62         if (tx.isActive) { tx.rollback }
63         pm.close
64       }
65     }
66   }
67
68   def getImageHtmlByName(name: String): Option[ImageHtml] = getByName[ImageHtml](name, classOf[ImageHtml])
69   def getImageCssByName(name: String): Option[ImageCss] = getByName[ImageCss](name, classOf[ImageCss])
70   def getImageIrcByName(name: String): Option[ImageIrc] = getByName[ImageIrc](name, classOf[ImageIrc])
71
72   private def getByName[T](name: String, c: java.lang.Class[T]): Option[T] = {
73     val pm = PMF.get.getPersistenceManager
74     try {
75       val query = pm.newQuery(c)
76       query.setUnique(true)
77       query.setFilter("name == n")
78       query.declareParameters("String n")
79       val result: T = query.execute(name).asInstanceOf[T]
80       if (result != None.orNull) { Some(result) } else { None }
81     } catch {
82       case me: MongoException => None
83     } finally {
84       pm.close
85     }
86   }
87
88   def getAllImageInfo: Option[List[ImageInfo]] = {
89     val pm = PMF.get.getPersistenceManager
90     try {
91       val query = pm.newQuery(classOf[ImageInfo])
92       query.setRange(0, QUERY_RANGE)
93       query.setOrdering("creation desc")
94       val result = query.execute.asInstanceOf[java.util.List[ImageInfo]]
95       Some(result.toList)
96     } catch {
97       case me: MongoException => None
98     } finally {
99       pm.close
100     }
101   }
102 }
