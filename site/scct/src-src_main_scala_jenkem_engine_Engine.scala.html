202     lazy val current: String = list(i)
203     lazy val last: String = list(i - 1)
204     lazy val next: String = list(i + 1)
205     lazy val isCurrentEqual: Boolean = current.equals(thisOne)
206     def isLastEqual: Boolean = thisOne.equals(last)
207     def isNextEqual: Boolean = thisOne.equals(next)
208     def isLastDark: Boolean = Pal.isDark(par.characters, last)
209     def isNextDark: Boolean = Pal.isDark(par.characters, next)
210     def isLastBright: Boolean = Pal.isBright(par.characters, last)
211     def isNextBright: Boolean = Pal.isBright(par.characters, next)
212     def isLastBrightOrEqual: Boolean = isLastBright || isLastEqual
213     def isNextBrightOrEqual: Boolean = isNextBright || isNextEqual
214     def leftCondition: Boolean = darkLeft && isLastDark && isNextBrightOrEqual
215     def rightCondition: Boolean = !darkLeft && isNextDark && isLastBrightOrEqual
216     def notLeftAndNotRight: Boolean = !leftCondition && !rightCondition
217     if (isFirstOrLast(i, range) || !isCurrentEqual || notLeftAndNotRight) { current } else { to }
218   }
