14 package jenkem.util
15
16 import jenkem.engine.color.Color
17
18 object ImageUtil {
19   def makeGreyPixel(imageRgb: Color.RgbMap, x: Int, y: Int): Short =
20     (getPixelRgbSum(imageRgb, x, y) / 3).shortValue
21
22   private def getPixelRgbSum(imageRgb: Color.RgbMap, x: Int, y: Int): Short =
23     imageRgb.get((y, x)) match {
24       case Some(rgb) => (rgb._1 + rgb._2 + rgb._3).shortValue
25       case _ => 0.shortValue
26     }
27
28   def getPixels(imageRgb: Color.RgbMap, x: Int, y: Int): Color.Rgb =
29     imageRgb.get((y, x)) match {
30       case Some(rgb) => (rgb._1, rgb._2, rgb._3)
31       case _ => (0.shortValue, 0.shortValue, 0.shortValue)
32     }
33 }
