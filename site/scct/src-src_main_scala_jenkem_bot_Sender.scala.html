14 package jenkem.bot
15
16 import jenkem.bot.status.Disconnected
17 import jenkem.bot.status.NotSending
18 import jenkem.bot.status.Connected
19 import jenkem.bot.status.Sending
20 import jenkem.bot.status.BotStatus
21
22 object Sender {
23   var botStatus = new BotStatus.Value(Disconnected, NotSending, "", "", "", IrcSettings.defaultDelayMs)
24   var stopSwitch = false
25   var isPlaying = false
26   var playThread: Thread = None.orNull
27
28   /**
29    * Runs a thread to play the image into the channel.
30    * this is done in Jenkems own thread so an asynchronous command can set the stop switch while the image is forwarded to IRC.
31    * Pircbots OutputMessage queue is thus bypassed by directly using sendRawLine method.
32    * the tread stops when the stopSwitch is changed to true or when the image is done being played.
33    */
34   class ImageSender(bot: JenkemBot, fullImage: List[String]) extends Runnable {
35     override def run: Unit = {
36       stopSwitch = false
37       isPlaying = true
38       botStatus = new BotStatus.Value(Connected, Sending, bot.getServer, bot.lastChan, bot.getNick, bot.getDelay)
39       val sendMe = "PRIVMSG " + bot.getChannels.head + " :"
40       sendImageLine(fullImage)
41       def sendImageLine(image: List[String]): Unit = {
42         if (!image.isEmpty) {
43           bot.sendRawLine(sendMe + image.head)
44           try {
45             Thread.sleep(bot.getMessageDelay)
46           } catch {
47             case ie: InterruptedException => bot.sendRawLine(sendMe + ie.getMessage)
48           }
49           if (!stopSwitch) { sendImageLine(image.tail) } else { resetStop(bot) }
50         } else { resetStop(bot) } //finished
51       }
52     }
53   }
54
55   /**
56    * Takes a String[] and floods it to IRC by using a new Thread.
57    * @param channel name of the channel
58    * @param out a String[] with the image as ASCII for IRC.
59    */
60   @throws(classOf[InterruptedException])
61   def playImage(bot: JenkemBot, out: List[String]): String = {
62     this.synchronized {
63       if (isPlaying) { "Bot is busy." }
64       else if (bot.getChannels.isEmpty) { "Bot is not in any channel." }
65       else {
66         playThread = new Thread(new ImageSender(bot, out))
67         playThread.start
68         "Playing image..."
69       }
70     }
71   }
72
73   def makeStop: Unit = if (isPlaying) {
74     stopSwitch = true
75     isPlaying = false
76   }
77
78   def resetStop(bot: JenkemBot): Unit = {
79     stopSwitch = false
80     isPlaying = false
81     botStatus = new BotStatus.Value(Connected, NotSending, bot.getServer, bot.lastChan, bot.getNick, bot.getDelay)
82   }
83 }
