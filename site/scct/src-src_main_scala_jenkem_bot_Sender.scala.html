16    * Runs a thread to play the image into the channel.
17    * this is done in Jenkems own thread so an asynchronous command can set the stop switch while the image is forwarded to IRC.
18    * Pircbots OutputMessage queue is thus bypassed by directly using sendRawLine method.
19    * the tread stops when the stopSwitch is changed to true or when the image is done being played.
20    */

22     override def run: Unit = {
23       stopSwitch = false
24       isPlaying = true
25       botStatus = new BotStatus.Value(Connected, Sending, bot.getServer, bot.lastChan, bot.getNick, bot.getDelay)
26       val sendMe = "PRIVMSG " + bot.getChannels.head + " :"
27       sendImageLine(fullImage)
28       def sendImageLine(image: List[String]): Unit = {
29         if (!image.isEmpty) {
30           bot.sendRawLine(sendMe + image.head)
31           try {
32             Thread.sleep(bot.getMessageDelay)
33           } catch {
34             case ie: InterruptedException => bot.sendRawLine(sendMe + ie.getMessage)
35           }
36           if (!stopSwitch) { sendImageLine(image.tail) } else { resetStop(bot) }
37         } else { resetStop(bot) } //finished
38       }
39     }
40   }
