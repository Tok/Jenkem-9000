291   private def generate(url: String, cs: ConversionSettings): List[String] = {
292     val invert = false
293     val originalImage = AwtImageUtil.bufferImage(url, "white", invert)
294     val originalWidth = originalImage.getWidth
295     val originalHeight = originalImage.getHeight
296     val (width, height) = Proportion.getWidthAndHeight(cs.proportion, cs.method, cs.width, originalWidth, originalHeight)
297     val scaled = AwtImageUtil.getScaled(originalImage, width, height, cs.kick, 0, 0)
298     val imageRgb = AwtImageUtil.getImageRgb(scaled)
299     val lastIndex = height
300     val params = cs.getParams(imageRgb)
301     def generate0(index: Int): List[String] = {
302       if (index + 2 > lastIndex) { Nil }
303       else { Engine.generateLine(params, index) :: generate0(index + 2) }
304     }
305     val hasColor = params.method.hasColor
306     val notPwntari = !params.method.equals(Method.Pwntari)
307     val colorString = if (hasColor) { ", Scheme: " + cs.schemeName } else { "" }
308     val powerString = if (notPwntari) { ", Power: " + params.power } else { "" }
309     val charsString = if (notPwntari) { ", Chars: " + params.characters } else { "" }
310     val widthDivisor = if (!params.method.equals(Method.Pwntari)) { 2 } else { 1 }
311     val message = List("Mode: " + params.method + colorString + powerString
312       + charsString + ", Width: " + (width / widthDivisor).intValue.toString)
313     message ::: generate0(0)
314   }
