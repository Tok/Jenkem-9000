14 package jenkem.engine
15
16 import jenkem.util.InitUtil
17
18 sealed abstract class Proportion
19 object Proportion {
20   case object Optimize extends Proportion
21   case object Exact extends Proportion
22   val values = List[Proportion](Optimize, Exact)
23   def valueOf(name: String): Option[Proportion] = values.find(_.toString.equalsIgnoreCase(name))
24   def default: Proportion = Optimize
25   def getWidthAndHeight(prop: Proportion, method: Method, maxWidth: Int,
26       originalWidth: Int, originalHeight: Int): (Int, Int) = {
27     if (prop.equals(Proportion.Optimize)) {
28       InitUtil.calculateProportionalSize(method, maxWidth, originalWidth, originalHeight)
29     } else {
30       InitUtil.calculateNewSize(method, maxWidth, originalWidth, originalHeight)
31     }
32   }
33 }
