16 /**
17 * Represents the possible kicks for the conversion.
18 * Depending on the Kick, Methods that convert two rows or columns at a time
19 * will start at position 0 or 1 in rows (Y) or columns (X).
20 *   X
21 * Y +---------->
22 *   | ## ## ##
23 *   | ## ## ##
24 *   | ## ## ##
25 *   | ## ## ##
26 *   v
27 */
28 sealed abstract class Kick(val xOffset: Int, val yOffset: Int)
29
30 object Kick {
31   case object OFF extends Kick(0, 0)
32   case object X extends Kick(1, 0)
33   case object Y extends Kick(0, 1)
34   case object XY extends Kick(1, 1)
35   val values: List[Kick] = List(OFF, X, Y, XY)
36   def valueOf(name: String): Option[Kick] = values.find(_.toString.equalsIgnoreCase(name))
37   def default: Kick = OFF
38 }
