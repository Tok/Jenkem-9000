304   private def startConversion(prepareImage: Boolean, resize: Boolean): Unit = {
305     if (!conversionDisabled) {
306       conversionDisabled = true
307       imagePreparer.getUrl match {
308         case Some(url) =>
309           try {
310             val totalStart = (new Date).getTime
311             if (prepareImage) { doPrepareImage(url) }
312             if (resize) { doPrepareImageData(url) }
313             if (prepareImage) {
314               makeImageInits
315               conversionDisabled = false
316               startConversion(false, false)
317             } else {
318               val (conversionTime, htmlTime) = makeConversion
319               val totalEnd = (new Date).getTime
320               val totalTime = totalEnd - totalStart
321               val status = "Image converted in " + conversionTime + "ms. " +
322                     "HTML generated in " + htmlTime + "ms. " +
323                     "Total time " + totalTime + "ms."
324               imagePreparer.setStatus(status)
325               //System.gc
326             }
327           } catch {
328             case iioe: javax.imageio.IIOException => imagePreparer.setError("Cannot read image from URL.")
329             case e: Exception => imagePreparer.setError(e.getMessage)
330           }
331         case None => { }
332       }
333     }
334   }

336   private def makeConversion(): (Long, Long) = {
337     val chars = charTextField.getValue.replaceAll("[,0-9]", "")
338     val contrast = contrastLabel.getValue.toShort
339     val brightness = brightnessLabel.getValue.toShort
340     convData = new ConversionData(contrast, brightness, chars)

342     val params = new Engine.Params(

344       imageData.imageRgb,
345       ircColorSetter.getColorMap,
346       chars,
347       ps,
348       contrast,
349       brightness,
350       Power.valueOf(powerBox.getValue.toString).get
351     )
352     val conversionStart = (new Date).getTime
353     ircOutput = generateIrcOutput(params, imageData.height)
354     val conversionEnd = (new Date).getTime
355     val conversionDiff = conversionEnd - conversionStart
356     outputDisplay.addIrcOutput(ircOutput.map(_ + "\n"))
357     val htmlDiff = updateInline(imageData.method, ircOutput, imagePreparer.getName)
358     ircConnector.refresh
359     conversionDisabled = false
360     (conversionDiff, htmlDiff)
361   }
