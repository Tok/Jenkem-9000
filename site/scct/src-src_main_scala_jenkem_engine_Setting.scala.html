16 sealed abstract class Setting(val caption: String)
17
18 object Setting {
19   case object UPDOWN extends Setting("\"_")
20   case object LEFTRIGHT extends Setting("[]")
21   case object DBQP extends Setting("dbqp")
22   case object DIAGONAL extends Setting("\\ /")
23   case object VERTICAL extends Setting("|")
24   case object HORIZONTAL extends Setting("-")
25   val values = List(UPDOWN, LEFTRIGHT, DBQP, DIAGONAL, VERTICAL, HORIZONTAL)
26
27   val min = -100
28   val max = 100
29   val default = 0
30
31   type Pair = (Boolean, Int)
32   class Instance(val ud: Pair, val lr: Pair, val dbqp: Boolean, val diag: Boolean, val vert: Boolean, val hor: Boolean) {
33     def has(setting: Setting): Boolean = {
34       if (setting.equals(DBQP)) { dbqp }
35       else if (setting.equals(DIAGONAL)) { diag }
36       else if (setting.equals(VERTICAL)) { vert }
37       else if (setting.equals(HORIZONTAL)) { hor }
38       else { getPair(setting)._1 }
39     }
40     def get(setting: Setting): Int = getPair(setting)._2
41     private def getPair(setting: Setting): Pair = {
42       setting match {
43         case UPDOWN => ud
44         case LEFTRIGHT => lr
45         case _ => (false, default)
46       }
47     }
48   }
49
50   def getInitial(hasAnsi: Boolean): Instance = {
51     new Instance((true, default), (true, default), (!hasAnsi), (!hasAnsi), (false), (false))
52   }
