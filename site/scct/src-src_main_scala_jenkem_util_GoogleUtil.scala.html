14 package jenkem.util
15
16 import java.net.URL
17 import scala.io.Source
18 import scala.util.Random
19 import scala.io.Codec
20 import java.net.HttpURLConnection
21 import java.io.BufferedReader
22 import java.io.InputStreamReader
23 import java.nio.charset.MalformedInputException
24
25 object GoogleUtil {
26   val searchUrl = "http://images.google.com/images?q="
27   val attributes = "&num=255&start=0&safe=off"
28   val encoding = "latin1" //prevents encoding errors (don't use UTF-8)
29   val agent = "Jenkem/9000"
30
31   def getUrlForTerm(term: String): Option[String] = {
32     if (term.equals("")) { getUrlForTerm(makeRandomSearchTerm) }
33     else {
34       val imagePageUrl = new URL(searchUrl + term + attributes)
35       val conn = imagePageUrl.openConnection.asInstanceOf[HttpURLConnection]
36       conn.setConnectTimeout(5000)
37       conn.setRequestProperty("user-agent", agent)
38       try {
39         conn.connect
40         val page = Source.fromInputStream(conn.getInputStream, encoding).mkString
41         //getting images from google cache at gstatic.con instead of original
42         val images = page.split("src=\"").filter(_.startsWith("http://")).filter(_.contains("gstatic.com"))
43         val imgUrls = images.map(_.split("\"")(0))
44         Some(Random.shuffle(imgUrls.toList).head)
45       } catch {
46         case pokemon: Throwable => None
47       } finally {
48         conn.disconnect
49       }
50     }
51   }
52
53   private def makeRandomSearchTerm: String = {
54     val imageName = if(Random.nextBoolean) { "DSC0" } else { "IMG0" }
55     val number = Random.nextDouble * 10
56     imageName + number.toString.take(5).filterNot(".".contains(_))
57   }
58 }
