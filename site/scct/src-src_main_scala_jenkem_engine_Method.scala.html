16 sealed abstract class Method(val name: String, val hasColor: Boolean, val hasKick: Boolean)

18 object Method {
19   case object Vortacular extends Method("Vortacular", true, true)
20   case object Pwntari extends Method("Pwntari", true, false)
21   case object Plain extends Method("Plain", false, true)
22   case object Stencil extends Method("Stencil", false, true)
23   val values = List(Vortacular, Pwntari, Plain, Stencil)
24   def valueOf(name: String): Option[Method] = values.find(_.name.equalsIgnoreCase(name))
