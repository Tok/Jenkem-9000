14 package jenkem.util
15
16 import jenkem.engine.color.Scheme
17
18 object ColorUtil {
19   val BC = String.valueOf('\u0002') // bold character for IRC
20   val CC = String.valueOf('\u0003') // color character for IRC
21
22   /**
23    * Converts an IRC color to a CSS color.
24    * @param ircColor
25    * @return cssColor
26    */
27   def ircToCss(ircColor: String): String = ircToCss(ircColor.toShort)
28
29   /**
30    * Converts an IRC color to a CSS color.
31    * @param ircColor
32    * @return cssColor
33    */
34   def ircToCss(ircColor: Short): String =
35     Scheme.valuOfIrcColor(ircColor) match {
36       case Some(ircColor) => rgbToCss(ircColor.rgb)
37       case _ => "#000000"
38     }
39
40   /**
41    * Converts an RGB color to a CSS color.
42    * @param rgb
43    * @return cssColor
44    */
45   private def rgbToCss(rgb: (Short, Short, Short)): String =
46     "#" + toHex(rgb._1) + toHex(rgb._2) + toHex(rgb._3)
47
48   /**
49    * Converts the provided interger to hex.
50    * @param i
51    * @return hex
52    */
53   private def toHex(i: Int): String = {
54     val ret = Integer.toHexString(i)
55     if (ret.length == 1) { "0" + ret } else { ret }
56   }
57
58   def makePwnIrc(bot: Short, top: Short): String = CC + bot + "," + top + "▄"
59
60   private def commaSplit(s: String): Array[String] = s.split(",")
61   def getFg(s: String): Int = commaSplit(s)(0).tail.toInt
62   def getBgString(s: String): String = commaSplit(s)(1).takeWhile(_.isDigit)
63   def getBg(s: String): Int = getBgString(s).toInt
64 }
