39   type Crops = (Int, Int, Int, Int)
40   val iconSize = 32

42   val colorWhite = new java.awt.Color(255, 255, 255)
43   val colorBlack = new java.awt.Color(0, 0, 0)

49   def bufferImage(url: String, bg: String, invert: Boolean, crops: Crops): BufferedImage = {
50     val color = if (bg.equalsIgnoreCase("white")) {
51       if (invert) { colorBlack } else { colorWhite }
52     } else {
53       if (invert) { colorWhite } else { colorBlack }
54     }
55     val buffered = doBuffer(url, bg, color, crops)
56     if (invert) { new RescaleOp(-1F, 255F, None.orNull).filter(buffered, None.orNull) }
57     else { buffered }
58   }

60   private def doBuffer(url: String, bg: String, color: java.awt.Color, crops: Crops): BufferedImage = {
61     val image = ImageIO.read(new URL(url))
62     val xs = crops._1
63     val xe = crops._2
64     val ys = crops._3
65     val ye = crops._4
66     val cropX = (image.getWidth * xs) / 100
67     val cropY = (image.getHeight * ys) / 100
68     val cropW = (image.getWidth * (xe - xs)) / 100
69     val cropH = (image.getHeight * (ye - ys)) / 100
70     val buffered = new BufferedImage(cropW, cropH, BufferedImage.TYPE_INT_RGB)
71     val g2d = buffered.getGraphics.asInstanceOf[Graphics2D]
72     g2d.setBackground(color)
73     g2d.clearRect(0, 0, cropW, cropH)
74     g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF)
75     val subimage = image.getSubimage(cropX, cropY, cropW, cropH)
76     g2d.drawImage(subimage, None.orNull, None.orNull)
77     g2d.dispose
78     buffered
79   }

81   def getRgb(img: BufferedImage, x: Int, y: Int): Color.Rgb = {
82     val argb = img.getRGB(x.intValue, y.intValue)
83     (((argb >> 16) & 0xff).toShort, ((argb >> 8) & 0xff).toShort, ((argb) & 0xff).toShort)
84   }

86   def getImageRgb(scaled: BufferedImage): Color.RgbMap = {
87     val keys = for {
88        y <- 0 until scaled.getHeight
89        x <- 0 until scaled.getWidth
90     } yield { (y, x) }
91     keys.map(t => ((t._1, t._2), getRgb(scaled, t._2, t._1))).toMap
92   }

118     val decoder = new BASE64Decoder

128   def getScaled(img: BufferedImage, width: Int, height: Int, kick: Kick, b: Int, c: Int): BufferedImage = {
129     val resized = new BufferedImage(width, height, img.getType)
130     val g2d = resized.createGraphics
131     g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF)
132     g2d.drawImage(img,
133         0, 0, width, height,
134         0, 0, img.getWidth, img.getHeight,
135         None.orNull)
136     g2d.dispose
137     val wx = resized.getWidth - (2 * kick.xOffset)
138     val hx = resized.getHeight - (2 * kick.yOffset)
139     val sub = resized.getSubimage(kick.xOffset, kick.yOffset, wx, hx)
140     new LookupOp(brightnessLot(b), None.orNull).filter(sub, sub)
141     new LookupOp(contrastLot(c), None.orNull).filter(sub, sub)
142     sub
143   }

145   private def brightnessLot(b: Int): ShortLookupTable = {
146     def getValue(v: Int): Short = Math.max(0, Math.min(255, v + b)).toShort
147     val table: Array[Short] = (0 to 255).map(getValue(_)).toArray
148     new ShortLookupTable(0, table)
149   }
150
151   private def contrastLot(c: Int): ShortLookupTable = {
152     val cf = (c + 100F).toFloat / 100F
153     def getValue(v: Int): Short = Math.max(0, Math.min(255, v * cf)).toShort
154     val table: Array[Short] = (0 to 255).map(getValue(_)).toArray
155     new ShortLookupTable(0, table)
156   }
