47   private def doBuffer(url: String, bg: String, color: java.awt.Color, crops: Crops): BufferedImage = {
48     val image = ImageIO.read(new URL(url))
49     val xs = crops._1
50     val xe = crops._2
51     val ys = crops._3
52     val ye = crops._4
53     val cropX = (image.getWidth * xs) / 100
54     val cropY = (image.getHeight * ys) / 100
55     val cropW = (image.getWidth * (xe - xs)) / 100
56     val cropH = (image.getHeight * (ye - ys)) / 100
57     val buffered = new BufferedImage(cropW, cropH, BufferedImage.TYPE_INT_RGB)
58     val g2d = buffered.getGraphics.asInstanceOf[Graphics2D]
59     g2d.setBackground(color)

61     g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF)

63     g2d.drawImage(subimage, None.orNull, None.orNull)
