40 class Submitter(val eventRouter: EventRouter) extends HorizontalLayout {
41   val maxNameLength = 40
42   var disable = false
43   var disableTrigger = false
44
45   setSpacing(true)
46
47   val nameTextField = new TextField
48   nameTextField.setWidth(250 + "px")
49   nameTextField.setImmediate(true)
50   nameTextField.addFocusListener(new FocusListener {
51     override def focus(event: FocusEvent): Unit = nameTextField.selectAll
52   })
53   nameTextField.addValueChangeListener(new Property.ValueChangeListener {
54     override def valueChange(event: ValueChangeEvent): Unit = {
55       if (!disable) {
56         if (nameTextField.getValue.size > maxNameLength) {
57           disable = true
58           nameTextField.setValue(nameTextField.getValue.substring(0, maxNameLength))
59         }
60       }
61       disable = false
62     }
63   })
64   addComponent(nameTextField)
65   setComponentAlignment(nameTextField, Alignment.MIDDLE_LEFT)
66
67   val submitButton = new Button("Submit To Gallery")
68   submitButton.addClickListener(new Button.ClickListener {
69     override def buttonClick(event: ClickEvent): Unit = {
70       eventRouter.fireEvent(new SaveImageEvent)
71     }
72   })
73   addComponent(submitButton)
74   setComponentAlignment(submitButton, Alignment.MIDDLE_LEFT)
75
76   val imageLayout = new GridLayout
77   imageLayout.setRows(1)
78   imageLayout.setColumns(1)
79   addComponent(imageLayout)
80   setComponentAlignment(imageLayout, Alignment.MIDDLE_LEFT)
81
82   val inversionCaption = new Label("Invert: ")
83   val inversionBox = new CheckBox
84   inversionBox.setImmediate(true)
85   inversionBox.addValueChangeListener(new Property.ValueChangeListener {
86     override def valueChange(event: ValueChangeEvent) {
87       if (!disableTrigger) {
88         eventRouter.fireEvent(new DoConversionEvent(true, true))
89       }
90     }
91   })
92   val invLayout = new HorizontalLayout
93   invLayout.addComponent(inversionCaption)
94   invLayout.addComponent(inversionBox)
95   addComponent(invLayout)
96   setComponentAlignment(invLayout, Alignment.MIDDLE_LEFT)
97
98   val bgOptions = List("white", "black")
99   val bgSelect = new OptionGroup(None.orNull, bgOptions)
100   bgSelect.setDescription("Only relvant for images with transparency.")
101   bgSelect.addStyleName("horizontal")
102   bgSelect.setNullSelectionAllowed(false)
103   bgSelect.setImmediate(true)
104   bgSelect.setValue(bgOptions(0))
105   bgSelect.addValueChangeListener(new Property.ValueChangeListener {
106     override def valueChange(event: ValueChangeEvent): Unit = {
107       if (!disableTrigger) {
108         eventRouter.fireEvent(new DoConversionEvent(true, true))
109       }
110     }
111   })
112   addComponent(bgSelect)
113   setComponentAlignment(bgSelect, Alignment.MIDDLE_LEFT)
114
115   def addIcon(img: BufferedImage): Unit = {
116     val image = new Image
117     image.setSource(AwtImageUtil.makeVaadinResource(img, "Icon"))
118     image.setWidth("32px")
119     image.setHeight("32px")
120     imageLayout.removeAllComponents
121     imageLayout.addComponent(image, 0, 0)
122   }
123   def setName(name: String): Unit = {
124     val clean = if (name.contains('.')) { name.take(name.lastIndexOf(".")) }
125     else { name }
126     if (clean.size > maxNameLength) {
127       nameTextField.setValue(clean.substring(0, maxNameLength))
128     } else {
129       nameTextField.setValue(clean)
130     }
131   }
132   def reset: Unit = {
133     disableTrigger = true
134     bgSelect.setValue(bgOptions(0))
135     inversionBox.setValue(false)
136     disableTrigger = false
137   }
138   def isInvert: Boolean = inversionBox.getValue
139   def getBg: String = bgSelect.getValue.toString
140   def getName: String = nameTextField.getValue
141   def enableSubmission(enabled: Boolean): Unit = submitButton.setEnabled(enabled)
142 }
