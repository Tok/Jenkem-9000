14 package jenkem.servlet
15
16 import jenkem.persistence.data.ImageHtml
17 import javax.jdo.annotations.PersistenceCapable
18 import javax.servlet.http.HttpServletRequest
19 import javax.servlet.http.HttpServletResponse
20 import jenkem.util.HtmlUtil
21
22 /**
23 * Servlet to retrieve and return stored HTML.
24 */
25 class HtmlOutputServlet extends AbstractOutputServlet {
26   override def doGet(request: HttpServletRequest, response: HttpServletResponse): Unit = {
27     response.setCharacterEncoding(encoding)
28     response.setContentType("text/html")
29     jenkemService.getImageHtmlByName(obtainName(request)) match {
30       case Some(imageHtml) =>
31         Option(imageHtml.html) match {
32           case Some(html) => response.getWriter.write(html)
33           case None => response.getWriter.write("Fail: HTML is empty.")
34         }
35       case None => response.getWriter.write("Fail: HTML couldn't be obtained.")
36     }
37   }
38 }
