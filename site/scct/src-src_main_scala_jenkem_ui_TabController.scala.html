14 package jenkem.ui
15
16 import scala.collection.immutable.ListMap
17
18 import com.vaadin.event.EventRouter
19 import com.vaadin.server.Page
20 import com.vaadin.ui.TabSheet
21 import com.vaadin.ui.TabSheet.SelectedTabChangeEvent
22
23 import jenkem.event.SaveImageEvent
24 import jenkem.ui.tab.GalleryTab
25 import jenkem.ui.tab.InfoTab
26 import jenkem.ui.tab.MainTab
27
28 class TabController(val eventRouter: EventRouter) {
29   var isReady = false
30   val defaultUrl = "http://upload.wikimedia.org/wikipedia/commons/0/03/RGB_Colorcube_Corner_White.png"
31   val tabSheet = new TabSheet
32   val mainTab = new MainTab(eventRouter)
33   val galleryTab = new GalleryTab(eventRouter)
34   val infoTab = new InfoTab
35   val tabs = ListMap(
36     mainTab.getCaption.toLowerCase -> mainTab,
37     galleryTab.getCaption.toLowerCase -> galleryTab,
38     infoTab.getCaption.toLowerCase -> infoTab)
39
40   def getTabSheet(page: Page): TabSheet = {
41     tabSheet.setWidth("1024px")
42     tabSheet.addSelectedTabChangeListener(new TabSheet.SelectedTabChangeListener() {
43       override def selectedTabChange(event: SelectedTabChangeEvent): Unit = {
44         if (isReady) {
45           val tabsheet = event.getTabSheet
46           val tab = tabsheet.getTab(tabsheet.getSelectedTab)
47           Option(tab) match {
48             case Some(tab) =>
49               if (tab.getCaption.toLowerCase.equals("main") && !mainTab.hasLink) {
50                 page.setUriFragment(tab.getCaption.toLowerCase + "/" + defaultUrl)
51                 mainTab.setLink(defaultUrl)
52               } else {
53                 page.setUriFragment(tab.getCaption.toLowerCase)
54               }
55             case None => { }
56           }
57         }
58       }
59     })
60     tabs.foreach((t) => tabSheet.addTab(t._2, t._2.getCaption))
61     tabSheet
62   }
63
64   def selectTab(frag: String): Unit = {
65     def selectMainWithDefault(): Unit = {
66       tabSheet.setSelectedTab(mainTab)
67       mainTab.setLink(defaultUrl)
68     }
69     Option(frag) match {
70       case Some(frag) =>
71         val split = frag.split("/", 2)
72         tabs.get(split(0)) match {
73           case None => selectMainWithDefault
74           case Some(tab) =>
75             tabSheet.setSelectedTab(tab)
76             if (tab.equals(mainTab)) {
77               if (split.length > 1) { mainTab.setLink(split(1)) }
78               else { mainTab.setLink(defaultUrl) }
79             }
80         }
81       case None => selectMainWithDefault
82     }
83   }
84
85   eventRouter.addListener(classOf[SaveImageEvent],
86     new {
87       def save: Unit = {
88         mainTab.saveImage
89         galleryTab.update
90       }
91     }, "save")
92 }
