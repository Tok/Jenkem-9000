14 package jenkem.engine.color
15
16 sealed abstract class Power(val name: String, val exponent: Float)
17
18 object Power {
19   case object Linear extends Power("Linear", 1F)
20   case object Quadratic extends Power("Quadratic", 2F)
21   case object Cubic extends Power("Cubic", 3F)
22   case object Quartic extends Power("Quartic", 4F)
23   val values: List[Power] = List(Linear, Quadratic, Cubic, Quartic)
24   def valueOf(name: String): Option[Power] = values.find(_.name.equalsIgnoreCase(name))
25   def default: Power = Linear
26 }
